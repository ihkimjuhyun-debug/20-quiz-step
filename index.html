function startDialAnimation() { const container = document.getElementById('cascade-container'); const words = ['Apple','Tiger','Ocean','Pizza','Robot','Smart']; const interval = setInterval(() => { const el = document.createElement('div'); el.className = 'cascade-word'; el.textContent = words[Math.floor(Math.random()*words.length)]; container.appendChild(el); setTimeout(() => el.remove(), 500); }, 100); setTimeout(async () => { clearInterval(interval); container.style.display = 'none'; document.getElementById('dial-final').classList.add('active'); try { const res = await fetch(`${API_BASE_URL}/get-word`, { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify({lang:'en'}) }); if(!res.ok) { const errData = await res.json().catch(() => ({})); throw new Error(errData.error || `네트워크 상태 코드: ${res.status}`); } const data = await res.json(); secretWord = data.word; secretCategory = data.category; setTimeout(() => { document.getElementById('phase-dial').classList.remove('active'); document.getElementById('phase-play').classList.add('active'); }, 1000); } catch(e) { // 이제 화면에 숫자만 뜨는 게 아니라 서버에서 보낸 정확한 한글 원인이 뜹니다. alert(`단어 생성 실패:\n\n${e.message}\n\nVercel 배포 상태와 Anthropic 크레딧을 확인해주세요.`); // 테스트를 위해 에러가 나도 임시 단어로 넘어갈 수 있게 우회(Bypass) 로직 추가 secretWord = "elephant"; secretCategory = "animal"; document.getElementById('phase-dial').classList.remove('active'); document.getElementById('phase-play').classList.add('active'); } }, 3000); }
